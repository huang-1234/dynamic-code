import React from "react";

const useLayoutEffectDemo = () => {
  React.useEffect(() => {
    console.log("effect:useEffect");
  }, []);
  React.useLayoutEffect(() => {
    console.log("effect:useLayoutEffect");
  }, []);
  return {
    node: <div>useLayoutEffectDemo</div>,
  };
};


function LayoutEffectDemo() {
  React.useEffect(() => {
    console.log("effect:useEffect");
  }, []);
  React.useLayoutEffect(() => {
    console.log("effect:useLayoutEffect");
  }, []);
  return (
    <div>useLayoutEffectDemo</div>
  )
}

export default LayoutEffectDemo
